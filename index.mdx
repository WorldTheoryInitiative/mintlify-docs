## Quick Examples

### Fetch Trending Markets

<CodeGroup>

```bash cURL
curl -X GET "https://api.polyrouter.io/functions/v1/markets-v2?platform=polymarket&limit=5" \
  -H "X-API-Key: YOUR_API_KEY"
```

```javascript JavaScript
const API_KEY = 'YOUR_API_KEY';

async function getMarkets() {
  const response = await fetch(
    'https://api.polyrouter.io/functions/v1/markets-v2?platform=polymarket&limit=5',
    { headers: { 'X-API-Key': API_KEY } }
  );
  
  const data = await response.json();
  data.markets.forEach(market => {
    console.log(`${market.title}`);
    console.log(`  Yes: ${(market.current_prices.yes.price * 100).toFixed(1)}%`);
    console.log(`  24h Volume: $${market.volume_24h.toLocaleString()}\n`);
  });
}

getMarkets();
```

```python Python
import requests

API_KEY = 'YOUR_API_KEY'

response = requests.get(
    'https://api.polyrouter.io/functions/v1/markets-v2',
    params={'platform': 'polymarket', 'limit': 5},
    headers={'X-API-Key': API_KEY}
)

for market in response.json()['markets']:
    print(f"{market['title']}")
    print(f"  Yes: {market['current_prices']['yes']['price'] * 100:.1f}%")
    print(f"  24h Volume: ${market['volume_24h']:,.0f}\n")
```

### Get NFL Betting Odds

<CodeGroup>

```bash cURL
# Find today's games
curl "https://api.polyrouter.io/functions/v1/list-games-v1?league=nfl&status=not_started&limit=5" \
  -H "X-API-Key: YOUR_API_KEY"

# Get odds for a specific game
curl "https://api.polyrouter.io/functions/v1/games-v1/BUFvKC20251020" \
  -H "X-API-Key: YOUR_API_KEY"
```

```javascript JavaScript
// Find today's games
async function getTodaysGames() {
  const response = await fetch(
    'https://api.polyrouter.io/functions/v1/list-games-v1?league=nfl&status=not_started&limit=5',
    { headers: { 'X-API-Key': API_KEY } }
  );
  return (await response.json()).games;
}

// Get markets for a game
async function getGameOdds(gameId) {
  const response = await fetch(
    `https://api.polyrouter.io/functions/v1/games-v1/${gameId}`,
    { headers: { 'X-API-Key': API_KEY } }
  );
  
  const game = (await response.json()).games[0];
  console.log(`${game.title}\n`);
  
  game.markets.forEach(platform => {
    const moneyline = platform.outcomes.find(o => 
      o.metadata?.market_type === 'moneyline'
    );
    if (moneyline) {
      console.log(`${platform.platform}: ${moneyline.name} ${(moneyline.price * 100).toFixed(1)}%`);
    }
  });
}

const games = await getTodaysGames();
if (games[0]) await getGameOdds(games[0].polyrouter_id);
```

```python Python
# Find today's games
def get_todays_games():
    response = requests.get(
        'https://api.polyrouter.io/functions/v1/list-games-v1',
        params={'league': 'nfl', 'status': 'not_started', 'limit': 5},
        headers={'X-API-Key': API_KEY}
    )
    return response.json()['games']

# Get markets for a game
def get_game_odds(game_id):
    response = requests.get(
        f'https://api.polyrouter.io/functions/v1/games-v1/{game_id}',
        headers={'X-API-Key': API_KEY}
    )
    
    game = response.json()['games'][0]
    print(f"{game['title']}\n")
    
    for platform in game['markets']:
        for outcome in platform['outcomes']:
            if outcome.get('metadata', {}).get('market_type') == 'moneyline':
                print(f"{platform['platform']}: {outcome['name']} {outcome['price'] * 100:.1f}%")

games = get_todays_games()
if games:
    get_game_odds(games[0]['polyrouter_id'])
```

</CodeGroup>

<Info>
  **PolyRouter Game IDs** follow the format `{AwayTeam}v{HomeTeam}{YYYYMMDD}` (e.g., `BUFvKC20251020`). Use `/list-games-v1` to discover IDs!
</Info>

### Query Price History

<CodeGroup>

```bash cURL
curl "https://api.polyrouter.io/functions/v1/price-history-v2?market_ids=516710&start_ts=1757036253&end_ts=1757270253&interval=1h&limit=24" \
  -H "X-API-Key: YOUR_API_KEY"
```

```javascript JavaScript
async function getPriceHistory(marketId, hours = 24) {
  const endTs = Math.floor(Date.now() / 1000);
  const startTs = endTs - (hours * 3600);
  
  const response = await fetch(
    `https://api.polyrouter.io/functions/v1/price-history-v2?market_ids=${marketId}&start_ts=${startTs}&end_ts=${endTs}&interval=1h&limit=${hours}`,
    { headers: { 'X-API-Key': API_KEY } }
  );
  
  const data = await response.json();
  console.log(`Price history for last ${hours} hours:\n`);
  
  data.data.forEach(point => {
    const date = new Date(point.timestamp * 1000).toISOString();
    console.log(`${date}: ${(point.price.close * 100).toFixed(1)}%`);
  });
}

getPriceHistory('516710', 24);
```

```python Python
import time

def get_price_history(market_id, hours=24):
    end_ts = int(time.time())
    start_ts = end_ts - (hours * 3600)
    
    response = requests.get(
        'https://api.polyrouter.io/functions/v1/price-history-v2',
        params={
            'market_ids': market_id,
            'start_ts': start_ts,
            'end_ts': end_ts,
            'interval': '1h',
            'limit': hours
        },
        headers={'X-API-Key': API_KEY}
    )
    
    print(f"Price history for last {hours} hours:\n")
    for point in response.json()['data']:
        print(f"{point['timestamp']}: {point['price']['close'] * 100:.1f}%")

get_price_history('516710', 24)
```
